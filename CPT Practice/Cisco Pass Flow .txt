enable                     # Enter privileged EXEC mode from user mode
configure terminal         # Enter global configuration mode from privileged EXEC mode
enable secret ccna2        # Set the encrypted enable password to 'ccna2'
service password-encryption # Encrypt all plain text passwords in config
line console 0             # Enter console line configuration mode
password cisco             # Set console password to 'cisco'
login                      # Require login for console access
exit                       # Exit console line config mode back to global config mode
no ip domain-lookup         # Disable DNS lookup (to prevent delays on mistyped commands)
end                        # Exit global config mode back to privileged EXEC mode
copy running-config startup-config  # Save the running config to NVRAM
